```julia:./list-tags.jl
#hideall
using JSON
data = JSON.parsefile("_data/tutorials.json")
tags = String[]
pkgs = String[]
for entry in data
  global tags, pkgs
  tags = union(tags, entry["tags"])
  pkgs = union(pkgs, entry["pkgs"])
end
tags = sort(unique(tags))
pkgs = sort(unique(pkgs))

println("""
~~~

</button>
~~~""")
```
